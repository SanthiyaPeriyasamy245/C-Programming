#include<stdio.h>
#include<math.h>
int main()
{
    int a[10];
    int i;
    for(i=0;i<10;i++)
    {
       a[i]=rand()%26+1;

    }
    for(i=0;i<10;i++)
    {
       printf("%d\n",a[i]);
    }
    while(i) //infinite loop it will stop by using break statement.
    {
       int swapped=0;
    {
        for(i=0;i<10;i++)
    {
       if(a[i]>a[i+1])
    {
           int temp=a[i];
           a[i]=a[i+1];
           a[i+1]=temp;
           swapped=1;
       }
    }
    if(swapped==0)
        break;
    }
}
printf("sorted list:");
 for(i=0;a[i]<a[10];i++)
    {
       printf("%d\n",a[i]);

    }
return 0;
}
